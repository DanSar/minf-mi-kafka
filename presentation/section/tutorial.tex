\begin{lstlisting}[]
$ tar -xzf kafka_2.11-1.0.0.tgz
$ cd kafka_2.11-1.0.0
\end{lstlisting}

\begin{lstlisting}[]
$ bin/zookeeper-server-start.sh config/zookeeper.properties
$ bin/kafka-server-start.sh config/server.properties
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]

\begin{lstlisting}[]

\end{lstlisting}
